function render_main_view() {

    switch(selected_tab) {
      case 0:
        render_vehicle_view(document.getElementById("timetable"));
        break;
      case 1:

        break;
      case 2:

        break;

        render_vehicle_view(document.getElementById("timetable"));
        break;
    }
}
